def initialize(name, x, y, zlist, width, length)

def execgl()
  if @movCoolDownTimer > 0
    @movCoolDownTimer = @movCoolDownTimer - 1
  elsif(@movCoolDownTimer == 0)
    @hitbox.y = @hitbox.y + 1
    @hitbox.x = @hitbox.x - 1
    @movCoolDownTimerH = 2
  end
end
