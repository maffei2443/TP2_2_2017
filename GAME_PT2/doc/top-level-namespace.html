def write(nome, table)
  nome.gsub!(/[\n\t]/, '')	# Retira caracteres teminais (\n por exemplo) e tabs

  ranking = File.open(nome, 'w+')

  t = Array.new
  i = 0
  table.each do |score, nome2|
    t[i] = [score, nome2]
    i += 1
  end
  t.sort! {|a,b| a <=> b}
  t.reverse!
  t.each{
    |x, y|
    ranking.puts "#{x} #{y}"
  }
  ranking.close
  return t
end
