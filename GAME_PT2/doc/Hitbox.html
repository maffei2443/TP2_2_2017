def check_hit(box)
  heightimp = false
  @zlist.each do |heightLevel|
    box.zlist.each do |hitheight|
      heightimp = true if heightLevel == hitheight
    end
  end
  if ((@x > box.x + box.width || box.x > @x + @width) || (@y > box.y + box.length || box.y > @y + @length)) || (heightimp == false)
    return false
  else
    return true
  end
end
