def update_frame
  if @movCoolDownTimer > 0
    @movCoolDownTimer -= 1
  elsif @movCoolDownTimer.zero?
    @hitbox.y = @hitbox.y + 1
    @hitbox.x = @hitbox.x - 1
    @movCoolDownTimer = 1
  end
end
